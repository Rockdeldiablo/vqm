<pre name='i_53_in' id='i_53_in' class='IFT'>ListArgColorPlot[list_List] :=
   Module[{xvars,hues,values},
         xvars	= Range[Length[list]];
         hues	= Hue[Arg[#]/(2 Pi)]&amp; /@ list;
         values	= Abs[list];
         fillit[xvars,hues,values]
         ]
